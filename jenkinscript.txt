pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                script {
                    git branch: 'mavenproject', url: 'https://github.com/manjumanjrekar/ownthesky_1.git/'
                }
            }
        }

        stage('Clean') {
            steps {
                script {
                    sh 'mvn clean'
                }
            }
        }

        stage('Test') {
            steps {
                script {
                    sh 'mvn test'
                }
            }
        }

        stage('Install') {
            steps {
                script {
                    sh 'mvn clean install'
             post {
                                always {
                                        junit 'target/surefire-reports/*.*xml'
                                        archiveArtifacts artifacts: 'target/*mm.jar', followSymlinks: false

                }
            }
        }
    }
}
}
}
